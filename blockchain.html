contract RockPaperScissors {

    enum Choice { None, Rock, Paper, Scissors }

    struct Player {

        address payable playerAddress;

        bytes32 commitment;

        Choice choice;

        uint256 wager;

    }

    //Part 3: State Variables:

    Player public player1;

    Player public player2;

    bool public isPlayer1Committed = false;

    bool public isPlayer2Committed = false;

    bool public isPlayer1Revealed = false;

    bool public isPlayer2Revealed = false;

    //Part 7: Modifiers - Defining Game Modifiers:

   

    modifier hasNotCommitted {

        require(msg.sender != player1.playerAddress || !isPlayer1Committed, "Player 1 already committed");

        require(msg.sender != player2.playerAddress || !isPlayer2Committed, "Player 2 already committed");

        _;

    }

    modifier bothPlayersCommitted {

        require(isPlayer1Committed && isPlayer2Committed, "Both players have not committed");

        _;

    }

    modifier bothPlayersRevealed {

        require(isPlayer1Revealed && isPlayer2Revealed, "Both players have not revealed");

        _;

    }

    modifier onlyPlayers {

        require(msg.sender == player1.playerAddress || msg.sender == player2.playerAddress, "Not a player");

        _;

    }

    //Part 4: Commit phase - Implementing the Commit Function:

    function commit(Choice _choice, string memory _secret) external payable hasNotCommitted {

        require(msg.value > 0, "Wager must be greater than 0");

        bytes32 _commitment = keccak256(abi.encodePacked(_choice, _secret));

        if (msg.sender == player1.playerAddress) {

            player1.commitment = _commitment;

            player1.wager = msg.value;

            isPlayer1Committed = true;

        } else {

            player2.commitment = _commitment;

            player2.wager = msg.value;

            isPlayer2Committed = true;

        }

    }

   

    //Part 5: Reveal Phase - Implementing the Reveal Function:

    function reveal(Choice _choice, string memory _secret) external onlyPlayers {

        require(keccak256(abi.encodePacked(_choice, _secret)) == (msg.sender == player1.playerAddress ? player1.commitment : player2.commitment), "Invalid reveal!");

        if (msg.sender == player1.playerAddress) {

            player1.choice = _choice;

            isPlayer1Revealed = true;

        } else {

            player2.choice = _choice;

            isPlayer2Revealed = true;

        }

    }

    //Part 6: Determining Winners - Implementing the Determine Winner Function:

    function determineWinner() external bothPlayersRevealed {

        // Compare choices and determine winner

        if (player1.choice == player2.choice) {

            // Draw, return wagers

            player1.playerAddress.transfer(player1.wager);

            player2.playerAddress.transfer(player2.wager);

        } else if (

            (player1.choice == Choice.Rock && player2.choice == Choice.Scissors) ||

            (player1.choice == Choice.Paper && player2.choice == Choice.Rock) ||

            (player1.choice == Choice.Scissors && player2.choice == Choice.Paper)

        ) {

            // Player 1 wins

            player1.playerAddress.transfer(player1.wager + player2.wager);

        } else {

            // Player 2 wins

            player2.playerAddress.transfer(player1.wager + player2.wager);

        }

        // Reset game state for the next round

        resetGameState();

    }

    function resetGameState() internal {

        isPlayer1Committed = false;

        isPlayer2Committed = false;

        isPlayer1Revealed = false;

        isPlayer2Revealed = false;

        delete player1;

        delete player2;

    }

}
